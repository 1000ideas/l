    • Argumenty:

      • search - szukna fraza

    # File app/models/l/gallery.rb, line 45
def self.search(search)
  find :all,
    :joins => :translations,
    :conditions =>
    ['(galleries_translations.name LIKE :pattern OR galleries_translations.content LIKE :pattern) AND locale = :locale',
    {
      :pattern => "%#{search}%",
      :locale => I18n.locale
    }
  ]
end
